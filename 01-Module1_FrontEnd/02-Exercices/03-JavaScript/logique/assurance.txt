  INITIALISER age, anneesPermis, nbAccidents, anneesClient à 0

  SINON
    SI nbAccidents == 0 ALORS

      tarif <- "rouge"

function tarif(age = 0, anneesPermis = 0, nbAccidents = 0, anneesClient = 0) {
  let tarif = "";

  age = parseInt(prompt("Entrez l'âge du conducteur :"));
  anneesPermis = parseInt(prompt("Entrez le nombre d'années de détention du permis :"));
  nbAccidents = parseInt(prompt("Entrez le nombre d'accidents responsables :"));
  anneesClient = parseInt(prompt("Entrez le nombre d'années en tant que client :"));

  switch (true) {
    case age < 25 && anneesPermis < 2:
      tarif = nbAccidents === 0 ? "rouge" : "Refusé";
      break;
    case (age < 25 && anneesPermis >= 2) || (age >= 25 && anneesPermis < 2):
      switch (nbAccidents) {
        case 0:
          tarif = "orange";
          break;
        case 1:
          tarif = "rouge";
          break;
        default:
          tarif = "Refusé";
      }
      break;
    default:
      switch (nbAccidents) {
        case 0:
          tarif = "vert";
          break;
        case 1:
          tarif = "orange";
          break;
        case 2:
          tarif = "rouge";
          break;
        default:
          tarif = "Refusé";
      }
  }

  if (tarif === "Refusé") {
    console.log(tarif);
    return;
  }

  if (anneesClient > 5) {
    switch (tarif) {
      case "rouge":
        tarif = "orange";
        break;
      case "orange":
        tarif = "vert";
        break;
      case "vert":
        tarif = "bleu";
        break;
    }
  }
  else {
    if (nbAccidents == 0) {
      tarif = "vert";
    }
    else if (nbAccidents == 1) {
      tarif = "orange";
    }
    else if (nbAccidents == 2) {
      tarif = "rouge";
    }
    else {
      console.log("Refusé");
      return;
    }
  }

  if (anneesClient > 5) {
    if (tarif == "rouge") {
      tarif = "orange";
    }
    else if (tarif == "orange") {
      tarif = "vert";
    }
    else if (tarif == "vert") {
      tarif = "bleu";
    }
  }

  console.log("Le tarif pour ce client est : " + tarif);
}
tarif();
